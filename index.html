async function buscar() { const termo = document.getElementById("busca").value.trim(); const area = document.getElementById("resultado"); if (!termo) { alert("Digite algo para pesquisar."); return; } area.innerHTML = "🎮 Buscando ofertas..."; try { const api = "https://api.mercadolibre.com/sites/MLB/search?q=" + encodeURIComponent(termo) + "&limit=12"; const resposta = await fetch(api); const dados = await resposta.json(); if (!dados.results || dados.results.length === 0) { area.innerHTML = "Nenhum produto encontrado."; return; } const lista = dados.results .sort((a, b) => a.price - b.price) .slice(0, 9); area.innerHTML = ""; lista.forEach((item, index) => { const card = document.createElement("a"); card.href = item.permalink; card.target = "_blank"; card.className = "card bg-gray-800 p-5 rounded-xl"; card.innerHTML = `

${item.title}

R$ ${item.price.toFixed(2)}

${index === 0 ? `
🏆 Melhor Preço
` : ``} `; area.appendChild(card); }); } catch (e) { console.error(e); area.innerHTML = "❌ Erro ao buscar preços. Tente novamente."; } }
